152.  Maximum contiguous product subarray ^q8H01qgK

public int maxProduct(int[] nums) {    
    
    int max = nums[0];
    int minCurrent = nums[0];
    int maxCurrent = nums[0];
    
    for(int i = 1; i < nums.length; i++){
        
        int temp = maxCurrent;
        maxCurrent = Math.max(nums[i],Math.max(minCurrent * nums[i], maxCurrent * nums[i]));
        minCurrent = Math.min(nums[i],Math.min(minCurrent * nums[i],temp * nums[i]));
        
        max = Math.max(max,maxCurrent);
    }
    return max;
} ^4Nohb89j
